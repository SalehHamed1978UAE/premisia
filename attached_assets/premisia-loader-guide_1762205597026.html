ðŸ’¡ Common Use Cases

1. Page Load Animation

<body data-page-loader>
    <!-- Page automatically shows loader until loaded -->
</body>

2. Button Loading State

// Add loader to button click
document.getElementById('generateBtn').addEventListener('click', async function() {
    const restore = PremisiaLoader.addButtonLoader(this, 'orbit');
    
    // Your async operation
    await generateProgram();
    
    // Restore button
    restore();
});

3. Async Operations

// Wrap any async function with loader
const result = await PremisiaLoader.withLoader(
    async () => {
        return await fetch('/api/analyze');
    },
    'hexagon',
    'Analyzing strategic vision...'
);

4. Progress Steps

// Show multi-step progress
const progress = PremisiaLoader.showProgressLoader([
    'Gathering requirements...',
    'Analyzing market data...',
    'Building strategy...',
    'Generating program...'
]);

// Move through steps
await step1();
progress.nextStep();

await step2();
progress.nextStep();

// Complete
progress.complete();

Strategic Analysis Loading:

async function analyzeStrategy(input) {
    const loaderId = PremisiaLoader.showFullPageLoader(
        'fractal',
        'AI is analyzing your strategic vision...'
    );
    
    try {
        const result = await premisiaAPI.analyze(input);
        PremisiaLoader.hideLoader(loaderId);
        return result;
    } catch (error) {
        PremisiaLoader.hideLoader(loaderId);
        showError(error);
    }
}
